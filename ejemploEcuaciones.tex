\section{Problema 1}

$f(n) - 9*f(n-2) = 5*3^n $

$h(n) = c_1*3^n + c_2*(-3)^n$

$p(n) = c_3*n*3^n$

\begin{align*}
 c_3* n* 3^n -9c_3*(n-2)*3^{n-2} &= 5*3^n  \\
 c_3* n* 3^n - c_3* n* 3^n + 2*c_3* 3^n  &= 5*3^n \\
 2*c_3* 3^n  &= 5*3^n \\
 c_3 &= 2.5
\end{align*}

$f(n) = c_1*3^n + c_2*(-3)^n + 2.5*n*3^n$
